37double PID_ruslan::compute(double setpoint, double measured) noexcept {
38 const double error = setpoint - measured;
39
40 // Integral component with anti-windup
41 integral = std::clamp(integral + error * dt, integral_min, integral_max);
42
43 // Derivative component
44 double derivative = 0.0;
45 if (prev_error_initialized) {
46 derivative = (error - prev_error) * inv_dt;
47 } else {
48 prev_error_initialized = true;
49 }
50 prev_error = error;
51
52 // PID output (with clamping)
53 const double output = Kp * error + Ki * integral + Kd * derivative;
54 return std::clamp(output, output_min, output_max);
55}
